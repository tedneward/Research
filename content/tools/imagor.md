Docker: `docker run -p 8000:8000 --rm -v $(pwd)/imagor-studio-data:/app/data -v ~/Pictures:/app/gallery -e DATABASE_URL="sqlite:///app/data/imagor-studio.db" shumc/imagor-studio`
